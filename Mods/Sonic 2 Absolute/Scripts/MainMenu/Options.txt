function display.menu
	if inputPress.buttonL == 1
		object.state = sound.menu
		menu.options.highlight = 0
	end if
	if inputPress.buttonR == 1
		object.state = gameplay.menu
		menu.options.highlight = 0
	end if
	switch menu.options.highlight
		case 0 //Display Highlighted
			if inputPress.left == 1
				object.state = datamang.menu
			end if
			if inputPress.right == 1
				object.state = gameplay.menu
			end if
			break
			case 1 // Fullscreen
				temp0 = inputPress.left
				temp0 |= inputPress.right
				if temp0 == 1
					saved.fullscreen ^= 1
					CallNativeFunction2(SetWindowFullScreen, saved.fullscreen, 0)
				end if
				break
			case 2 // Screen Width		
				if inputPress.left == 1
					switch screen.xsize
						case 320
							CallNativeFunction2(SetScreenWidth, 384, 0)
							break
						case 384
							CallNativeFunction2(SetScreenWidth, 426, 0)
							break
						case 426
							CallNativeFunction2(SetScreenWidth, 320, 0)
							break
						default
							CallNativeFunction2(SetScreenWidth, 426, 0)
							break
					end switch
				end if
				if inputPress.right == 1
					switch screen.xsize
						case 320
							CallNativeFunction2(SetScreenWidth, 426, 0)
							break
						case 384
							CallNativeFunction2(SetScreenWidth, 320, 0)
							break
						case 426
							CallNativeFunction2(SetScreenWidth, 384, 0)
							break
						default
							CallNativeFunction2(SetScreenWidth, 426, 0)
							break
					end switch
				end if
				break
			case 3 // Window Scale		
				if inputPress.left == 1					
					saved.scale--
					if saved.scale < 1
						saved.scale = 4
					end if
					CallNativeFunction2(SetWindowScale, saved.scale, 0)
				end if
				if inputPress.right == 1
					saved.scale++
					if saved.scale > 4
						saved.scale = 1
					end if
					CallNativeFunction2(SetWindowScale, saved.scale, 0)
				end if
				break
			case 4 // Graphics Mode
				temp0 = 0
				if inputPress.left == 1
					temp0 = 1
					saved.graphics--
					if saved.graphics < 0
						saved.graphics = 2
					end if
				end if
				if inputPress.right == 1
					temp0 = 1
					saved.graphics++
					if saved.graphics > 2
						saved.graphics = 0
					end if
				end if
				if temp0 == 1
					CallFunction(ResetPal)
					LoadPalette("MainMenu.act", 0, 0, 0, 256)
					LoadPalette("SaveSlot.act", 1, 0, 0, 256)
					LoadPalette("AchieveIcon.act", 2, 0, 0, 256)
					CallFunction(BlackandWhite)
					CreateTempObject(TypeName[Palette Police], saved.graphics, 0, 0)
				end if				
				break
			case 5 // VSync
				temp0 = inputPress.left
				temp0 |= inputPress.right
				if temp0 == 1
					saved.vsync ^= 1
					CallNativeFunction2(SetWindowVSync, saved.vsync, 0)
				end if
				break
			case 6 // Scaling Mode
				if inputPress.left == 1					
					saved.scalemode--
					if saved.scalemode < 0
						saved.scalemode = 3
					end if
					CallNativeFunction2(SetWindowScaleMode, saved.scalemode, 0)
				end if
				if inputPress.right == 1
					saved.scalemode++
					if saved.scalemode > 3
						saved.scalemode = 0
					end if
					CallNativeFunction2(SetWindowScaleMode, saved.scalemode, 0)
				end if
				break
			case 7 // Refresh Rate
				temp0 = inputPress.left
				temp0 |= inputPress.right
				if temp0 == 1
					if saved.fps == 60
						saved.fps = 50
					else
						saved.fps = 60
					end if
					CallNativeFunction2(SetFrameRate, saved.fps, 0)
				end if
				break
		//none of the display options seem to work, does the exe not have those yet?
	end switch
	
	CheckEqual(inputPress.left, 1)
	temp0 = checkResult
	CheckEqual(inputPress.right, 1)
	temp0 |= checkResult
	CheckEqual(inputPress.buttonL, 1)
	temp0 |= checkResult
	CheckEqual(inputPress.buttonR, 1)
	temp0 |= checkResult
	if temp0 == 1
		CallFunction(main.sound)
		WriteSaveRAM()
	end if
	
	if inputPress.down == 1
		CallFunction(main.sound)
		menu.options.highlight++
		menu.options.highlight %= 8
	end if
	if inputPress.up == 1
		CallFunction(main.sound)
		menu.options.highlight--
		while menu.options.highlight < 0
			menu.options.highlight += 8
		loop
	end if
	
	temp0 = menu.options.highlight
	temp0 *= options.gap
	menu.sub.ymax = temp0
	temp0 -= 180
	menu.sub.ymin = temp0
	
	if menu.sub.ytarget < menu.sub.ymin
		menu.sub.ytarget = menu.sub.ymin
	else
		if menu.sub.ytarget > menu.sub.ymax
			menu.sub.ytarget = menu.sub.ymax
		end if
	end if
	
	if inputPress.buttonB == 1
		object[9].priority = PRIORITY_ACTIVE
		object[9].value14 = 0
		object[9].value9 = 0
		object[9].state = Menu.firstcontrol
	end if
end function

	end if
	if inputPress.up == 1
