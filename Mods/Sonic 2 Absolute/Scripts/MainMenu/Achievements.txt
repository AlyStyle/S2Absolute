function Modes.LoadOut
	if menu.fade < 320
		menu.fade += 16
		music.volume -= 2
	else
		StopMusic()
		LoadStage()
	end if
	SetScreenFade(0, 0, 0, menu.fade)
end function

event ObjectMain				
	CallFunction(object.state)
end event

function AchieveControl
	if menu.sub.ytarget < menu.sub.ymin
		menu.sub.ytarget = menu.sub.ymin
	else
		if menu.sub.ytarget > menu.sub.ymax
			menu.sub.ytarget = menu.sub.ymax
		end if
	end if
	
	if inputPress.up == 1
		PlaySfx(SfxName[Score Add], 0)
		achieve.highlight--
		while achieve.highlight < 0
			achieve.highlight += achieve.count
		loop
	else
		if inputPress.down == 1
			PlaySfx(SfxName[Score Add], 0)
			achieve.highlight++
			achieve.highlight %= achieve.count
		end if
	end if
	
	if inputPress.buttonB == 1
		object[9].priority = PRIORITY_ACTIVE
		object[9].value14 = 0
		object[9].value9 = 0
		object[9].state = Menu.firstcontrol
	end if
	
	temp0 = achieve.highlight
	temp0 *= achieve.gap
	menu.sub.ymax = temp0
	temp0 -= 130
	menu.sub.ymin = temp0
end function

event ObjectDraw
	if object[9].value9 != 0
		if menu.zigzag.yoffset != 240
			 menu.zigzag.yoffset += 10
		end if
	else
		if menu.zigzag.yoffset != 0
			 menu.zigzag.yoffset -= 10
		end if
	end if
	CallFunction(AchieveDraw)
end event

function AchieveDraw
	temp0 = screen.ycenter
	temp0 += 125
	temp0 -= menu.zigzag.yoffset
	temp0 <<= 1
	
	temp0 -= menu.sub.yoffset
	
	DrawSpriteScreenXY(4, screen.xcenter, temp0)
	temp0 += 12
	DrawSpriteScreenXY(195, screen.xcenter, temp0)
	temp1 = screen.xcenter
	temp1 += 32
	DrawNumbers(69, temp1, temp0, achieve.compcount, 2, 10, 0)	
	temp1 += 30
	DrawNumbers(69, temp1, temp0, achieve.count, 2, 10, 0)	
	temp0 += options.gap
	
	temp5 = 0
	while temp5 < achieve.count
			GetTableValue(temp1, temp5, AchieveOrder)
			arrayPos1 = 400
			arrayPos1 += temp1			
			if temp5 == achieve.highlight
				//Draw Highlight
				SetActivePalette(1, 0, screen.ysize)
				DrawSpriteScreenXY(196, screen.xcenter, temp0)
				SetActivePalette(0, 0, screen.ysize)
				
				temp2 = temp1
				temp2 += 237
				DrawSpriteScreenXY(temp2, screen.xcenter, temp0)
			else
				//Draw Unhighlight
				temp2 = temp1
				temp2 += 197
				DrawSpriteScreenXY(temp2, screen.xcenter, temp0)
			end if
			DrawSpriteScreenXY(194, screen.xcenter, temp0)
			CallFunction(AchieveDescribe)
			temp2 = temp1
			temp2 += 277
			if saveRAM[arrayPos1] == 1
				SetActivePalette(2, 0, screen.ysize)
			else
				SetActivePalette(3, 0, screen.ysize)
			end if
			DrawSpriteScreenXY(temp2, screen.xcenter, temp0)
			SetActivePalette(0, 0, screen.ysize)
			

			
	temp0 += achieve.gap
	temp5++
	loop
end function

		math.blue = temp1

			math.green *= weight.green
			math.blue *= weight.blue

			temp1 += math.green
			temp1 += math.red
			temp2 = weight.red
			temp2 += weight.green
			temp2 += weight.blue
			temp1 /= temp2
			
		else
			temp1 = math.blue
			temp1 += math.green
			temp1 += math.red
			temp1 /= 3
		
		end if
		
		temp1 &= 0xFF
		temp2 = temp1
		temp2 <<= 8
		temp2 += temp1
		temp2 <<= 8
		temp2 += temp1
		
		SetPaletteEntry(3, temp0, temp2)
		temp0++
	loop
end event
